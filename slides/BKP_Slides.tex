\section{Background}\label{background}

\begin{frame}{Background}
\begin{itemize}
\item
  Estimating a \textbf{continuous probability function} from
  \textbf{binary} or \textbf{binomial observations} is fundamental in
  statistics and machine learning.
\item
  Applications include:

  \begin{itemize}
  \item
    Binary classification \citep{MacKenzie2014, Wen2025KLR}
  \item
    Probability calibration
    \citep{Sung2020binaryCalibration, Dimitriadis2023}
  \item
    Relative abundance modeling \citep{Martin2020}
  \item
    Longitudinal patient-reported outcomes
    \citep{Najera2018HRQoL, Najera2019PRO}
  \item
    Infectious disease risk prediction \citep{Diggle2007Loa}
  \item
    \ldots{}
  \end{itemize}
\end{itemize}
\end{frame}

\begin{frame}{Classical methods \& Challenges}
\protect\phantomsection\label{classical-methods-challenges}
\begin{itemize}
\item
  \textbf{Parametric method}: logistic regression is
  \textcolor{red}{computationally efficient} but
  \textcolor{blue}{limited in capturing complex nonlinear patterns}.
\item
  \textbf{Nonparametric method}: Gaussian Process (GP) offers
  \textcolor{red}{flexibility and principled uncertainty quantification},
  but \textcolor{blue}{approximate inference} for binary
  (\emph{non-Gaussian}) likelihood increases computational cost.
\end{itemize}
\end{frame}

\begin{frame}{Beta Kernel Process}
\protect\phantomsection\label{beta-kernel-process}
\begin{itemize}
\item
  \textbf{Beta Kernel Process (BKP)}: a scalable and interpretable
  nonparametric model for binomial probability surfaces
  \citep{Goetschalckx:2011, MacKenzie2014, Rolland2019BKP}
\item
  \textbf{Key features}:

  \begin{itemize}
  \item
    \textcolor{red}{Computational efficiency}: closed-form posterior
    inference
  \item
    \textcolor{red}{More transparency and straightforward}:
    beta-binomial conjugate pair
  \item
    \textcolor{red}{Uncertainty quantification}: adaptive
    decision-marking
  \end{itemize}
\item
  \textbf{Dirichlet Kernel Process (DKP)}: a natural extension for
  categorical or multinomial data
\end{itemize}
\end{frame}

\section{Summary}\label{summary}

\begin{frame}{Summary}
\protect\phantomsection\label{summary-1}
\begin{itemize}
\item
  Developed the
  \textcolor{red}{\bf first publicly available \pkg{BKP} package}
  implementing BKP and DKP models
\item
  \textcolor{red}{\bf Methodological contributions:}

  \begin{itemize}
  \item
    Provided a \textbf{Bayesian interpretation} of the BKP model
  \item
    Proposed a \textbf{flexible framework for constructing priors},
    including data-adaptive options
  \item
    Developed a \textbf{robust and efficient hyperparameter optimization
    strategy} using LOOCV and multi-start reparameterization.
  \end{itemize}
\end{itemize}
\end{frame}

\begin{frame}{Future Development Directions}
\protect\phantomsection\label{future-development-directions}
\begin{itemize}
\item
  \textbf{Model extensions}: Extend the BKP framework to multivariate,
  functional, and spatio-temporal data, as well as mixed-type covariates
\item
  \textbf{Alternative likelihoods}: Explore likelihoods beyond the
  binomial family, e.g., negative binomial for over-dispersed counts,
  geometric for waiting-time modeling
\item
  \textbf{Community contributions}: We invite developers to contribute
  via GitHub pull requests
  \href{https://github.com/Jiangyan-Zhao/BKP/pulls}{\texttt{https://github.com/Jiangyan-Zhao/BKP/pulls}}
\end{itemize}
\end{frame}
